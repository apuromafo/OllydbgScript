IsEXEorDLL:
 gmi eip, MODULEBASE
 mov modulebase, $RESULT, 4
 mov tmp, modulebase
 add tmp, 3C        // e_lfanew
 mov PEOffset, [tmp], 4 // Start of PE Header
 add PEOffset, modulebase
 mov tmp, PEOffset
 add tmp, 16    // Characteristics
 mov tmp, [tmp], 2
 shr tmp, 0D
 cmp tmp, 1
 je IS_DLL
 mov IS_DLL, 0
 mov ext, "exe"
 jmp end
 IS_DLL:
 mov IS_DLL, 1
 mov ext, "dll"
end:
ret
