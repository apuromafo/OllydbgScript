var t
mov t,esp
sub t,4

EOE Error
EOB Break
bphws t, "w"
run

Error:
esti
bphwc t
bphws t, "w"
run

Break:
bphwc t
sto
sto
cmt eip, "This is the entry point (OEP)"
ret
